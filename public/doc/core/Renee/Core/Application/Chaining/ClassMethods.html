def chain_method(*methods)
  methods.each do |m|
    class_eval <<-EOT, __FILE__, __LINE__ + 1
      alias_method :#{m}_without_chain, :#{m}
      def #{m}(*args, &blk)
        chain(blk) { |subblk| #{m}_without_chain(*args, &subblk) }
      end
      private :#{m}_without_chain
    EOT
  end
end
